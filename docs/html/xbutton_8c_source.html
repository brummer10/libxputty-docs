26Widget_t* add_button(Widget_t *parent, const char * label,
27 int x, int y, int width, int height) {
28
29 Widget_t *wid = create_widget(parent->app, parent, x, y, width, height);
30 wid->label = label;
31 wid->adj_y = add_adjustment(wid,0.0, 0.0, 0.0, 1.0,1.0, CL_TOGGLE);
32 wid->adj = wid->adj_y;
33 wid->scale.gravity = CENTER;
34 wid->func.expose_callback = _draw_base_button;
35 wid->func.enter_callback = os_transparent_draw;
36 wid->func.leave_callback = os_transparent_draw;
37 wid->func.button_press_callback = _button_pressed;
38 wid->func.button_release_callback = _button_released;
39 return wid;
40}
